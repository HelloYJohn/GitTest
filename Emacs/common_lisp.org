a function return the sum of less number n
#+BEGIN_SRC emacs-lisp
(defun sum (n)
  (let ((s 0))
    (dotimes (i n s)
      (incf s i))))

(sum 999)
#+END_SRC

#+BEGIN_SRC C -n -r includes <stdio.h>
int sum(int n) {
    int i, s = 0;
    for (i = 0; i < n; i++) {
        s += i;
    }
    return s;
}
printf("%d\n", sum(100));
#+END_SRC
